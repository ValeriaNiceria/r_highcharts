4 Coluna preenchida

Gráfico exemplo - Coluna preenchida 

library(tidyverse)
library(highcharter)

    highchart() %>%
        hc_add_series(
          data = dados_plot,
          type = "column",
          name = "Valor",
          pointPadding = 0.2,
          color = 'rgba(136, 145, 194, 1)',
          hcaes(
            x = as.character(ano), 
            y = valor1
          ),
          tooltip = list(pointFormat = '<span style="color:{series.color}">\u25CF</span>{series.name}: <b>{point.valor:.,2f}</b><br/>
      Valor1: <b>{point.valor1:.2f}%</b>
      ')
        ) %>% 
        hc_add_series(
          data = dados_plot,
          type = "column",
          name = "Valor1",
          pointPadding = 0.4,
          color = 'rgba(0, 7, 45,.9)',
          hcaes(
            x = as.character(ano), 
            y = valor
          )
        ) %>%
        hc_xAxis(categories = as.character(dados_plot$ano)) %>% 
        hc_yAxis(title = list(text = "Valor"), lineColor ="#f7f7f7",  gridLineColor="#f4f4f4") %>% 
        hc_plotOptions(
          column = list(
            grouping = FALSE,
            shadow = FALSE,
            borderWidth = 0
          ),
          line = list(dataLabels = list(enabled = TRUE))
        )
