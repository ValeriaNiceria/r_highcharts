2 Gráficos - Line e Spline

Gráfico exemplo - Line 

# Carregando os pacotes
library(tidyverse)
library(highcharter)

# Criando o gráfico
hchart(
  dados,
  "line",
  hcaes(
    x = as.character(date), 
    y = value, 
    group = grupo, 
    color = color
  )
) %>% 
hc_plotOptions(
  line = list(
  dataLabels = list(
    enabled = TRUE, 
    format = '{y:.,0f}'
    )
  )
) %>%
hc_yAxis(title = list(text = "Valores")) %>% 
hc_xAxis(title = list(text = "Meses")) %>% 
hc_title(text = "Gráfico Exemplo - Line") %>% 
hc_colors(as.character(unique(dados$color)))

Gráfico exemplo - Spline 

# Carregando os pacotes
library(tidyverse)
library(highcharter)

# Criando o gráfico
hchart(
  dados,
  "spline",
  hcaes(
    x = as.character(date), 
    y = value, 
    group = grupo, 
    color = color
  )
) %>% 
  hc_plotOptions(
    spline = list(
      dataLabels = list(
        enabled = TRUE, 
        format = '{y:.,0f}'
      )
    )
  ) %>%
  hc_yAxis(title = list(text = "Valores")) %>% 
  hc_xAxis(title = list(text = "Meses")) %>% 
  hc_title(text = "Gráfico Exemplo - Spline") %>% 
  hc_colors(as.character(unique(dados$color)))
