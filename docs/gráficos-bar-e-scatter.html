5 Gráficos - Bar e Scatter

Gráfico exemplo - Bar

# Carregando os pacotes
library(tidyverse)
library(highcharter)

# Criando o gráfico
hchart(
  dados,
  "bar",
  hcaes(
    x = as.character(date), 
    y = value, 
    group = grupo, 
    color = color
  )
) %>% 
  hc_plotOptions(
    bar = list(
      dataLabels = list(
        enabled = TRUE, 
        format = '{y:.,0f}'
      )
    )
  ) %>%
  hc_yAxis(title = list(text = "Valores")) %>% 
  hc_xAxis(title = list(text = "Meses")) %>% 
  hc_title(text = "Gráfico Exemplo - Bar") %>% 
  hc_colors(as.character(unique(dados$color)))

Gráfico exemplo - Bar Stacked

# Carregando os pacotes
library(tidyverse)
library(highcharter)

# Criando o gráfico
hchart(
  dados,
  "bar",
  hcaes(
    x = as.character(date), 
    y = value, 
    group = grupo, 
    color = color
  )
) %>% 
  hc_plotOptions(
    bar = list(
      stacking="normal",
      dataLabels = list(
        enabled = TRUE, 
        format = '{y:.,0f}'
      )
    )
  ) %>%
  hc_yAxis(title = list(text = "Valores")) %>% 
  hc_xAxis(title = list(text = "Meses")) %>% 
  hc_title(text = "Gráfico Exemplo - Bar Stacked") %>% 
  hc_colors(as.character(unique(dados$color))) %>% 
  hc_exporting(enabled = TRUE)
