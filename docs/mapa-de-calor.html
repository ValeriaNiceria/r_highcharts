3 Mapa de calor

Gráfico exemplo - Mapa de calor 

library(tidyverse)
library(highcharter)


 dados_plot %>% 
      hchart(
        type = 'heatmap', 
        hcaes(
          x = ano, 
          y = tipo, 
          value = pct
        ),
        name = "Teste"
      ) %>% 
      hc_plotOptions(
        series = list(
          dataLabels = list(
            enabled = TRUE, 
            format = '{point.value:.2f}%'
          )
        )
      ) %>% 
      hc_colorAxis(
        maxColor = '#222c6a',
        minColor = '#009761'
      ) %>% 
      hc_tooltip(
        pointFormat = paste('{point.tipo}: <b>{point.pct:.,2f}%</b><br>')
      ) %>% 
      hc_yAxis(title = list(text = 'Tipo')) %>%
      hc_xAxis(title = list(text = 'Ano'))  %>%
      hc_legend(enabled = FALSE)
